# 删除命名空间卡住一直不退出

```bash
kubectl delete namespace keda

kubectl get namespace keda -o yaml

# finalizer 阻止退出
spec:
  finalizers:
  - kubernetes

kubectl edit namespace keda
spec:
  finalizers: []
# 强制删除，上面的方法不生效
kubectl get namespace keda -o json | jq '.spec.finalizers=[]' | kubectl replace --raw /api/v1/namespaces/keda/finalize -f -
```
